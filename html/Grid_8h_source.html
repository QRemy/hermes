185  T interpolate(const Vector3d &position) const {
186  // position on a unit grid
187  Vector3d r = (position - gridOrigin) / spacing;
188 
189  // indices of lower and upper neighbors
190  int ix, iX, iy, iY;
191  if (reflective) {
192  reflectiveClamp(r.x, Nx, ix, iX);
193  reflectiveClamp(r.y, Ny, iy, iY);
194  } else {
195  periodicClamp(r.x, Nx, ix, iX);
196  periodicClamp(r.y, Ny, iy, iY);
197  }
198 
199  // linear fraction to lower and upper neighbors
200  double fx = r.x - floor(r.x);
201  double fX = 1 - fx;
202  double fy = r.y - floor(r.y);
203  double fY = 1 - fy;
204 
205  // bilinear interpolation (see
206  // https://en.wikipedia.org/wiki/Bilinear_interpolation#Unit_square)
207  T b(0.);
208  // f(0,0) (1 - x) (1 - y) +
209  b += get(ix, iy) * fX * fY;
210  // f(1,0) x (1 - y) +
211  b += get(iX, iy) * fx * fY;
212  // f(0,1) (1 - x) y +
213  b += get(ix, iY) * fX * fy;
