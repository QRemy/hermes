147 QArea BremsstrahlungGALPROP::dsdk_IntermediateEnergy(
148  const QNumber &gamma_i, const QNumber &gamma_f, const QNumber &p_i,
149  const QNumber &p_f, const QNumber &k, int Z) const {
150  QNumber L = 2. * log((gamma_i * gamma_f + p_i * p_f - 1_num) / k);
151  QNumber epsilon_i = log((gamma_i + p_i) / (gamma_i - p_i));
152  QNumber epsilon_f = log((gamma_f + p_f) / (gamma_f - p_f));
153 
154  QNumber ininner_factor =
155  epsilon_i * (gamma_i * gamma_f + pow<2>(p_i)) / pow<3>(p_i);
156  ininner_factor -=
157  epsilon_f * (gamma_i * gamma_f + pow<2>(p_f)) / pow<3>(p_f);
158  ininner_factor += 2. * k * gamma_i * gamma_f / pow<2>(p_i * p_f);
159 
160  QNumber inner_factor = 8. / 3. * gamma_i * gamma_f / p_i / p_f;
161  inner_factor += pow<2>(k) *
162  (pow<2>(gamma_i * gamma_f) + pow<2>(p_i * p_f)) /
163  pow<3>(p_i * p_f);
164  inner_factor += k / 2. / p_i / p_f * ininner_factor;
165 
166  QNumber factor = 4. / 3.;
167  factor -= 2. * gamma_i * gamma_f * (pow<2>(p_f) + pow<2>(p_i)) /
168  pow<2>(p_f * p_i);
169  factor +=
170  epsilon_i * gamma_f / pow<3>(p_i) + epsilon_f * gamma_i / pow<3>(p_f);
171  factor -= epsilon_i * epsilon_f / p_i / p_f;
172  factor += L * inner_factor;
173 
174  return pow<2>(Z * r_electron) * alpha_fine * (p_f / p_i) / k * factor;
175 }

197 QDiffCrossSection BremsstrahlungGALPROP::getDiffCrossSectionForTargetDirectly(
198  Target t, const QEnergy &T_electron, const QEnergy &E_gamma) const {
199  int Z, N;
200  if (t == Target::HII) {
201  Z = 1;
202  N = 0;
203  }
204  if (t == Target::HI) {
205  Z = 1;
206  N = 1;
207  }
208  if (t == Target::He) {
209  Z = 2;
210  N = 2;
211  }
212 
213  QNumber k = E_gamma / mc2_units;
214  QNumber T_electron_i = T_electron / mc2_units;
215  QNumber T_electron_f = T_electron_i - k;
216 
217  if (T_electron_f <= 0_num) return QDiffCrossSection(0);
218  if (T_electron_i < 0.01_MeV / mc2_units) return QDiffCrossSection(0);
219 
220  QNumber E_electron_i = T_electron_i + 1_num;
221  QNumber E_electron_f = T_electron_f + 1_num;
222 
223  QNumber p_i = sqrt(T_electron_i * (T_electron_i + 2_num));
224  QNumber p_f = sqrt(T_electron_f * (T_electron_f + 2_num));
225 
226  QNumber gamma_i = E_electron_i;
227  QNumber gamma_f = E_electron_f;
228 
229  QNumber beta_i = sqrt(1_num - 1. / (gamma_i * gamma_i));
230  QNumber beta_f = sqrt(1_num - 1. / (gamma_f * gamma_f));
231 
232  if (T_electron_i < 0.07_MeV / mc2_units) {
233  return ElwertFactor(beta_i, beta_f, Z) *
234  dsdk_LowEnergy(p_i, p_f, k, Z) / mc2_units;
235  }
236 
237  if (T_electron_i < 2.0_MeV / mc2_units) {
238  return ElwertFactor(beta_i, beta_f, Z) * xiFunc(T_electron_i, k, Z, N) *
239  dsdk_IntermediateEnergy(gamma_i, gamma_f, p_i, p_f, k, Z) /
240  mc2_units;
241  }
242 
243  return dsdk_HighEnergy(gamma_i, gamma_f, k, Z, N) / mc2_units;
244 }
