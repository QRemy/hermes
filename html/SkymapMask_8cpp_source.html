71  // std::cerr << "open = " << radian2degree(open) << " angle = " <<
72  // radian2degree(testAngle) << " close = " << radian2degree(close);
73  // If the opening and closing angle are the same, the full circle is allowed

75  // false
76  if (open == close) return true;
