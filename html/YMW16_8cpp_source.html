392  double slc, clc, sbc, cbc, xc, yc, zc, rgalc;
393  double rp, RR, xyp, zp;
394  double theta, alpha;
395  double Dmin = 1e5;
396 
397  // center of Gum Nebula
398  const double lc = 264;
399  const double bc = -4;
400  const double dc = 450;
401 
402  if (*m_5 >= 1) return 0;
403 
404  slc = std::sin(lc / RAD);
405  clc = std::cos(lc / RAD);
406  sbc = std::sin(bc / RAD);
407  cbc = std::cos(bc / RAD);
408 
409  rgalc = dc * cbc;
410  xc = rgalc * slc;
411  yc = R0 * 1000 - rgalc * clc;
412  zc = dc * sbc;
413 
414  theta = std::fabs(std::atan(
415  (zz - zc) / std::sqrt((xx - xc) * (xx - xc) + (yy - yc) * (yy - yc))));
416  zp = ((t5.Agn) * (t5.Agn) * (t5.Kgn)) /
417  std::sqrt(((t5.Agn) * (t5.Agn)) +
418  ((t5.Agn) * (t5.Agn) * (t5.Kgn) * (t5.Kgn)) /
419  (std::tan(theta) * std::tan(theta)));
420  xyp = zp / std::tan(theta);
421  if ((t5.Agn - std::fabs(xyp)) < 1e-15) {
422  alpha = pi / 2;
423  } else {
424  alpha =
425  -std::atan((-(t5.Agn) * (t5.Kgn) * xyp) /

583  double delta_l = -69.75 / RAD;

586  double Al = 3000; // pc

602  s_delta * sin(delta_l);

605  ss_dp = s_delta * cos(delta_l) -

609  Y = d * (cos(iag) * (ss_dp * cos(npag) - sc_dp * sin(npag)) +
610  c_dag * sin(iag)) -
611  D_l * sin(iag);
612  Z = d * (sin(iag) * (ss_dp * cos(npag) - sc_dp * sin(npag)) -
613  c_dag * cos(iag)) +
614  D_l * cos(iag);
615  Rl = sqrt(X * X + Y * Y);
616  if (Rl > mc * Al || fabs(Z) > mc * Hl) {
617  return 0;
618  } else {
619  gl = 1;
620  *w_lmc = 1;
621  }
622  return gl * (t9.nlmc) * std::exp(-(Rl * Rl) / (Al * Al)) *
623  std::pow(2 / (std::exp(-Z / Hl) + std::exp(Z / Hl)), 2);

662  c_cpl = std::cos(alpha_cp - alpha_l);

664  c_dag = c_delta * std::cos(delta_l) *

666  s_delta * std::sin(delta_l);
667  s_dag = std::sin(std::acos(c_dag));

672  //(xc,yc,zc,iag,npag)------get(X,Y,Z) is (x',y',z')
673  double X = d * (sc_dp * std::cos(npag) + ss_dp * std::sin(npag));
674  double Y =

676  c_dag * std::sin(iag)) -

681  D_l * std::cos(iag);

684  // ((alpha_D,delta_D,D_l)----((alpha_l,delta_l,D_l)-----get(dag_D,pag_D)

692  std::cos(delta_D) * std::sin(delta_l) * std::cos(alpha_D - alpha_l);
693

695  X_D = D_D * (sc_dpD * std::cos(npag) + ss_dpD * std::sin(npag));
696  Y_D = D_D * (std::cos(iag) *
697  (ss_dpD * std::cos(npag) - sc_dpD * std::sin(npag)) +
698  c_dagD * std::sin(iag)) -
699  D_l * std::sin(iag);

701  (ss_dpD * std::cos(npag) - sc_dpD * std::sin(npag)) -
702  c_dagD * std::cos(iag)) +
703  D_l * std::cos(iag);
704  R_D = std::sqrt((X - X_D) * (X - X_D) + (Y - Y_D) * (Y - Y_D) +
705  (Z - Z_D) * (Z - Z_D));
706  if (R_D > (mc * A_D)) {
707  return 0;
708  } else
709  gD = 1;
710 
711  return gD * (t10.n30D) * exp(-(R_D * R_D) / (A_D * A_D));
