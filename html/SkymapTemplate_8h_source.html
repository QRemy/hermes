247 template <typename QPXL, typename QSTEP>
248 void SkymapTemplate<QPXL, QSTEP>::compute() {
249  std::cout << "hermes::Integrator: Number of Threads: " << getThreadsNumber()
250  << std::endl;
251 
252  if (integrator == nullptr)
253  throw std::runtime_error(
254  "Provide an integrator with Skymap::setIntegrator()");
255 
256  // Generate cache tables in integrator for a given skymap parameter
257  if (integrator->isCacheTableEnabled()) {
258  integrator->setSkymapParameter(skymapParameter);
259  integrator->initCacheTable();
260  }
261 
262  // Progressbar init
263  progressbar =
264  std::make_shared<ProgressBar>(ProgressBar(getUnmaskedPixelCount()));
265  progressbar_mutex = std::make_shared<std::mutex>();
266  progressbar->setMutex(progressbar_mutex);
267  progressbar->start("Compute skymap");
268 
269  auto job_chunks = getThreadChunks(size());
270  std::vector<std::thread> threads;
271  for (auto &chunk : job_chunks) {
272  threads.push_back(
273  std::thread(&SkymapTemplate<QPXL, QSTEP>::computePixelRange, this,
274  chunk.first, chunk.second, integrator));
275  }
276  for (auto &t : threads) {
277  t.join();
278  }
