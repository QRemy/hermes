111 QDiffIntensity PiZeroIntegrator::integrateOverLOS(
112  const QDirection &direction_, const QEnergy &Egamma_) const {
113  QDiffIntensity total_diff_flux(0.0);
114 
115  auto gasType = ngdensity->getGasType();
116 
117  // Sum over rings
118  for (const auto &ring : *ngdensity) {
119  // TODO(adundovi): this could be checked better
120  if (!ngdensity->isRingEnabled(ring->getIndex())) continue;
121 
123  // p_Theta_f(r) = profile(r) * Theta_in(r)
124  auto p_Theta_f = [ring, gasType, this](const Vector3QLength &pos) {
125  return (ring->isInside(pos)) ? dProfile->getPDensity(gasType, pos)
126  : 0;
127  };
128  auto normIntegrand = [this, p_Theta_f,
129  direction_](const QLength &dist) {
130  return p_Theta_f(
131  getGalacticPosition(this->positionSun, dist, direction_));
132  };
133 
134  // optimize LOS integration limits:
135  // instead of 0 and getMaxDistance(dir)
136  auto b = ring->getBoundaries();
137  auto rho = positionSun.getRho();
138  QLength r_min = rho - b.second;
139  if (r_min < 0_m) r_min = 0_m;
140  QLength r_max = rho + b.second;
141  if (r_max > getMaxDistance(direction_))
142  r_max = getMaxDistance(direction_);
143 
144  QColumnDensity normIntegral =
145  gslQAGIntegration<QColumnDensity, QPDensity>(normIntegrand, r_min,
146  r_max, 200);
147 
148  // LOS is not crossing the current ring at all, skip
149  if (normIntegral == QColumnDensity(0)) continue;
150 
151  // std::cerr << "normIntegral" << normIntegral << std::endl;
152 
154  // los_f = emissivity(r) * profile(r) * Theta_in(r)
155  auto los_f = [ring, gasType, this](const Vector3QLength &pos,
156  const QEnergy &Egamma_) {
157  return (ring->isInside(pos))
158  ? dProfile->getPDensity(gasType, pos) *
159  this->integrateOverEnergy(pos, Egamma_)
160  : 0;
161  };
162  auto losIntegrand = [this, los_f, direction_,
163  Egamma_](const QLength &dist) {
164  return los_f(
165  getGalacticPosition(this->positionSun, dist, direction_),
166  Egamma_);
167  };
168  QDiffIntensity losIntegral =
169  simpsonIntegration<QDiffFlux, QGREmissivity>(losIntegrand, r_min,
170  r_max, 500) /
171  (4_pi * 1_sr);
172 
173  // Finally, normalize LOS integrals, separatelly for HI and CO
174  total_diff_flux +=
175  ring->getColumnDensity(direction_) / normIntegral * losIntegral;
176  }
177 
178  return total_diff_flux;
179 }

181 QPiZeroIntegral PiZeroIntegrator::integrateOverEnergy(
182  const Vector3QLength &pos_, const QEnergy &Egamma_) const {
183  if (cacheTableInitialized) {
184  return getIOEfromCache(pos_, Egamma_);
185  }

187  QPiZeroIntegral total(0);
188 
189  // TODO(adundovi): micro-optimization - E_min = E_gamma +
190  // m_pi^2c^4/(4E_gamma)
191  std::vector<QDiffCrossSection> diffCrossSectionVector;
192  std::transform(crList[0]->beginAfterEnergy(Egamma_), crList[0]->end(),
193  std::back_inserter(diffCrossSectionVector),
194  [this, Egamma_](const QEnergy &E) -> QDiffCrossSection {
195  return crossSec->getDiffCrossSection(E, Egamma_);
196  });
197 
198  for (const auto &crDensity : crList) {
199  auto pid_projectile = crDensity->getPID();
200 
201  std::vector<QPDensity> cosmicRayVector;
202  std::transform(crDensity->beginAfterEnergy(Egamma_), crDensity->end(),
203  std::back_inserter(cosmicRayVector),
204  [crDensity, pos_](const QEnergy &E) -> QPDensity {
205  return crDensity->getDensityPerEnergy(E, pos_) * E;
206  });
207 
208  std::vector<QPiZeroIntegral> integral;
209  std::transform(cosmicRayVector.begin(), cosmicRayVector.end(),
210  diffCrossSectionVector.begin(),
211  std::back_inserter(integral),
212  [](const QPDensity &a, const QDiffCrossSection &b) {
213  return a * b * c_light;
214  });
215  // log-integration
216  auto integralOverEnergy =
217  std::log(crDensity->getEnergyScaleFactor()) *
218  std::accumulate(integral.begin(), integral.end(),
219  QPiZeroIntegral(0));
220 
221  // std::cerr << "IOE = " <<
222  // crossSec->getDiffCrossSection(*crDensity->beginAfterEnergy(Egamma_),
223  // Egamma_) << std::endl;
224 
225  for (const auto &neutralGas : ngdensity->getAbundanceFractions()) {
226  auto pid_target = neutralGas.first;
227  auto f_target = neutralGas.second;

230  }
