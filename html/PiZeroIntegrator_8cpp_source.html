107 QDiffIntensity PiZeroIntegrator::integrateOverLOS(
108  const QDirection &direction_, const QEnergy &Egamma_) const {
109  QDiffIntensity total_diff_flux(0.0);
110 
111  // Sum over rings
112  for (const auto &ring : *ngdensity) {
113  // TODO(adundovi): this could be checked better
114  if (!ngdensity->isRingEnabled(ring->getIndex())) continue;
115 
116  // Normalization-part
117  auto normI_f = [ring, this](const Vector3QLength &pos) {
118  return (ring->isInside(pos)) ? this->densityProfile(pos) : 0;
119  };
120  auto normIntegrand = [this, normI_f, direction_](const QLength &dist) {
121  return normI_f(
122  getGalacticPosition(this->positionSun, dist, direction_));
123  };
124  QColumnDensity normIntegrals =
125  gslQAGIntegration<QColumnDensity, QPDensity>(
126  normIntegrand, 0, getMaxDistance(direction_), 500);
127 
128  // LOS is not crossing the current ring at all
129  if (normIntegrals == QColumnDensity(0)) continue;
130 
131  // Integral over emissivity
132  auto losI_f = [ring, this](const Vector3QLength &pos,
133  const QEnergy &Egamma_) {
134  return (ring->isInside(pos))
135  ? this->densityProfile(pos) *
136  this->integrateOverEnergy(pos, Egamma_)
137  : 0;
138  };
139  auto losIntegrand = [this, losI_f, direction_,
140  Egamma_](const QLength &dist) {
141  return losI_f(
142  getGalacticPosition(this->positionSun, dist, direction_),
143  Egamma_);
144  };
145  QDiffIntensity losIntegrals =
146  simpsonIntegration<QDiffFlux, QGREmissivity>(
147  losIntegrand, 0, getMaxDistance(direction_), 500) /
148  (4_pi * 1_sr);
149 
150  // Finally, normalize LOS integrals, separatelly for HI and CO
151  if (ngdensity->getRingType() == neutralgas::RingType::HI) {
152  total_diff_flux += (ring->getHIColumnDensity(direction_)) /
153  normIntegrals * losIntegrals;
154  }
155  if (ngdensity->getRingType() == neutralgas::RingType::CO) {
156  total_diff_flux += (2 * X0Function(Vector3QLength(0)) *
157  ring->getCOIntensity(direction_)) /
158  normIntegrals * losIntegrals;
159  }
160  }
161 
162  return total_diff_flux;
163 }

173 QPiZeroIntegral PiZeroIntegrator::integrateOverEnergy(
174  const Vector3QLength &pos_, const QEnergy &Egamma_) const {
175  if (cacheTableInitialized) return getIOEfromCache(pos_, Egamma_);
176 
177  QPiZeroIntegral total(0);
178 
179  // TODO(adundovi): micro-optimization - E_min = E_gamma +
180  // m_pi^2c^4/(4E_gamma)
181  std::vector<QDiffCrossSection> diffCrossSectionVector;
182  std::transform(crList[0]->beginAfterEnergy(Egamma_), crList[0]->end(),
183  std::back_inserter(diffCrossSectionVector),
184  [this, Egamma_](const QEnergy &E) -> QDiffCrossSection {
185  return crossSec->getDiffCrossSection(E, Egamma_);
186  });
187 
188  for (const auto &crDensity : crList) {
189  auto pid_projectile = crDensity->getPID();
190 
191  std::vector<QPDensity> cosmicRayVector;
192  std::transform(crDensity->beginAfterEnergy(Egamma_), crDensity->end(),
193  std::back_inserter(cosmicRayVector),
194  [crDensity, pos_](const QEnergy &E) -> QPDensity {
195  return crDensity->getDensityPerEnergy(E, pos_) * E;
196  });
197 
198  std::vector<QPiZeroIntegral> integral;
199  std::transform(cosmicRayVector.begin(), cosmicRayVector.end(),
200  diffCrossSectionVector.begin(),
201  std::back_inserter(integral),
202  [](const QPDensity &a, const QDiffCrossSection &b) {
203  return a * b * c_light;
204  });
205  // log-integration
206  auto integralOverEnergy =
207  std::log(crDensity->getEnergyScaleFactor()) *
208  std::accumulate(integral.begin(), integral.end(),
209  QPiZeroIntegral(0));
210 
211  for (const auto &neutralGas : ngdensity->getAbundanceFractions()) {
212  auto pid_target = neutralGas.first;
213  auto f_target = neutralGas.second;
214  total += f_target * crossSec->getSigma(pid_projectile, pid_target) *
215  integralOverEnergy;
216  }
217  }
218  return total;
219 }
