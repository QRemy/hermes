156 double Random::randPowerLaw(double index, double min, double max) {
157  if ((min < 0) || (max < min)) {
158  throw std::runtime_error(
159  "Power law distribution only possible for 0 <= min <= max");
160  }
161  // check for index -1!
162  if ((std::abs(index + 1.0)) < std::numeric_limits<double>::epsilon()) {
163  double part1 = log(max);
164  double part2 = log(min);
165  return exp((part1 - part2) * rand() + part2);
166  } else {
167  double part1 = std::pow(max, index + 1);
168  double part2 = std::pow(min, index + 1);
169  double ex = 1 / (index + 1);
170  return std::pow((part1 - part2) * rand() + part2, ex);
171  }
172 }

174 double Random::randBrokenPowerLaw(double index1, double index2,
175  double breakpoint, double min, double max) {
176  if ((min <= 0) || (max < min)) {
177  throw std::runtime_error(
178  "Power law distribution only possible for 0 < min <= max");
179  }
180  if (min >= breakpoint) {
181  return this->randPowerLaw(index2, min, max);
182  } else if (max <= breakpoint) {
183  return this->randPowerLaw(index2, min, max);
184  } else {
185  double intPL1;
186  // check if index1 = -1
187  if ((std::abs(index1 + 1.0)) < std::numeric_limits<double>::epsilon()) {
188  intPL1 = log(breakpoint / min);
189  } else {
190  intPL1 =
191  (std::pow(breakpoint, index1 + 1) - std::pow(min, index1 + 1)) /
192  (index1 + 1);
193  }
194  double intPL2;
195  // check if index2 = -1
196  if ((std::abs(index2 + 1.0)) < std::numeric_limits<double>::epsilon()) {
197  intPL2 =
198  log(max / breakpoint) * std::pow(breakpoint, index1 - index2);
199  } else {
200  intPL2 =
201  (std::pow(max, index2 + 1) - std::pow(breakpoint, index2 + 1)) *
202  std::pow(breakpoint, index1 - index2) / (index2 + 1);
203  }
204  if (rand() > intPL1 / (intPL1 + intPL2))
205  return randPowerLaw(index2, breakpoint, max);
206  else
207  return randPowerLaw(index1, min, breakpoint);
208  }
209 }

273 void Random::seed(uint32 *const bigSeed, const uint32 seedLength) {
274  initialize(19650218UL);
275  int i = 1;
276  uint32 j = 0;
277  int k = (N > seedLength ? N : seedLength);
278  for (; k; --k) {
279  state[i] =
280  state[i] ^ ((state[i - 1] ^ (state[i - 1] >> 30)) * 1664525UL);
281  state[i] += (bigSeed[j] & 0xffffffffUL) + j;
282  state[i] &= 0xffffffffUL;
283  ++i;
284  ++j;
285  if (i >= N) {
286  state[0] = state[N - 1];
287  i = 1;
288  }
289  if (j >= seedLength) j = 0;
290  }
291  for (k = N - 1; k; --k) {
292  state[i] =
293  state[i] ^ ((state[i - 1] ^ (state[i - 1] >> 30)) * 1566083941UL);
294  state[i] -= i;
295  state[i] &= 0xffffffffUL;
296  ++i;
297  if (i >= N) {
298  state[0] = state[N - 1];
299  i = 1;
300  }
301  }
302  state[0] = 0x80000000UL; // MSB is 1, assuring non-zero initial array
303  reload();
304 }

348 Random::uint32 Random::hash(time_t t, clock_t c) {
349  static uint32 differ = 0; // guarantee time-based seeds will change
350 
351  uint32 h1 = 0;
352  unsigned char *p = (unsigned char *)&t;
353  for (size_t i = 0; i < sizeof(t); ++i) {
354  h1 *= std::numeric_limits<unsigned char>::max() + 2U;
355  h1 += p[i];
356  }
357  uint32 h2 = 0;
358  p = (unsigned char *)&c;
359  for (size_t j = 0; j < sizeof(c); ++j) {
360  h2 *= std::numeric_limits<unsigned char>::max() + 2U;
361  h2 += p[j];
362  }
363  return (h1 + differ++) ^ h2;
364 }
