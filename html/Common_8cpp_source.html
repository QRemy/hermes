71  const QDirection &direction,
72  const QLength &galacticBorder,

74  static const Vector3QLength positionGC(0, 0, 0);
75  // static const QLength galacticBorder = 30_kpc; // for example JF12 is
76  // zero for r > 20kpc static const QLength zBorder = 5_kpc;
77  Vector3QLength vecSunToGalBorder;
78  vecSunToGalBorder.setRThetaPhi(1_m, direction[0], direction[1]);
79 
80  QLength a = (positionGC - positionSun).getR();
81  QLength c = galacticBorder;
82  QAngle gamma = vecSunToGalBorder.getAngleTo(positionSun - positionGC);
83 
84  QLength sphericalBorder =
85  a * cos(gamma) + sqrt(c * c - a * a * (1 - cos(2 * gamma)) / 2.0);
86  QLength heightBroder = fabs(zBorder / cos(direction[0]));
