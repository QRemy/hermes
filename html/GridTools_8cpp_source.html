86 void initTurbulence(std::shared_ptr<VectorGrid> grid, double Brms, double lMin,
87  double lMax, double alpha, int seed, bool helicity,
88  double H) {

90  size_t Ny = grid->getNy();
91  size_t Nz = grid->getNz();
92  if ((Nx != Ny) or (Ny != Nz))
93  throw std::runtime_error("turbulentField: only cubic grid supported");
94 
95  Vector3d spacing = grid->getSpacing();
96  if ((spacing.x != spacing.y) or (spacing.y != spacing.z))
97  throw std::runtime_error("turbulentField: only equal spacing suported");
98 
99  if (lMin < 2 * spacing.x)
100  throw std::runtime_error("turbulentField: lMin < 2 * spacing");
101  if (lMin >= lMax) throw std::runtime_error("turbulentField: lMin >= lMax");
102  if (lMax > Nx * spacing.x / 2)
103  throw std::runtime_error("turbulentField: lMax > size / 2");
104 
105  size_t n = Nx; // size of array
106  size_t n2 = (size_t)std::floor(n / 2) +
107  1; // size array in z-direction in configuration space
108 
109  // arrays to hold the complex vector components of the B(k)-field
110  fftwf_complex *Bkx, *Bky, *Bkz;
111  Bkx = (fftwf_complex*)fftwf_malloc(sizeof(fftwf_complex) * n * n * n2);
112  Bky = (fftwf_complex*)fftwf_malloc(sizeof(fftwf_complex) * n * n * n2);
113  Bkz = (fftwf_complex*)fftwf_malloc(sizeof(fftwf_complex) * n * n * n2);
114 
115  Random random;
116  if (seed != 0) random.seed(seed); // use given seed
117 
118  // calculate the n possible discrete wave numbers
119  double K[n];
120  for (int i = 0; i < n; i++) K[i] = (double)i / n - i / (n / 2);
121 
122  // construct the field in configuration space
123  int i;
124  double k;
125 
126  // only used if there is a helicity
127  double Bktot, Bkplus, Bkminus, thetaplus, thetaminus;
128 
129  // parameters goes for non helical calculations
130  double theta, phase, cosPhase, sinPhase;
131 
132  double kMin = spacing.x / lMax;
133  double kMax = spacing.x / lMin;
134  Vector3f b; // real b-field vector
135  Vector3f ek, e1, e2; // orthogonal base
136  Vector3f n0(1, 1, 1); // arbitrary vector to construct orthogonal base
137 
138  for (size_t ix = 0; ix < n; ix++) {
139  for (size_t iy = 0; iy < n; iy++) {
140  for (size_t iz = 0; iz < n2; iz++) {
141  i = ix * n * n2 + iy * n2 + iz;
142  ek.setXYZ(K[ix], K[iy], K[iz]);
143  k = ek.getR();
144 
145  // wave outside of turbulent range -> B(k) = 0
146  if ((k < kMin) || (k > kMax)) {
147  Bkx[i][0] = 0;
148  Bkx[i][1] = 0;
149  Bky[i][0] = 0;
150  Bky[i][1] = 0;
151  Bkz[i][0] = 0;
152  Bkz[i][1] = 0;
153  continue;
154  }
155 
156  // construct an orthogonal base ek, e1, e2
157  // (for helical fields together with the real
158  // transform the following convention must be
159  // used: e1(-k) = e1(k), e2(-k) = - e2(k)
160  if (helicity == true) {
161  if (ek.getAngleTo(n0) < 1e-3_rad) { // ek parallel to
162  // (1,1,1)
163  e1.setXYZ(-1, 1, 0);
164  e2.setXYZ(1, 1, -2);
165  } else { // ek not parallel to (1,1,1)
166  e1 = n0.cross(ek);
167  e2 = ek.cross(e1);
168  }
169  e1 /= e1.getR();
170  e2 /= e2.getR();
171 
172  double Bkprefactor =
173  static_cast<double>(mu0) / (4 * M_PI * std::pow(k, 3));
174  Bktot = fabs(random.randNorm() * std::pow(k, alpha / 2));
175  Bkplus = Bkprefactor * sqrt((1 + H) / 2) * Bktot;
176  Bkminus = Bkprefactor * sqrt((1 - H) / 2) * Bktot;
177  thetaplus = 2 * M_PI * random.rand();
178  thetaminus = 2 * M_PI * random.rand();
179  double ctp = cos(thetaplus);
180  double stp = sin(thetaplus);
181  double ctm = cos(thetaminus);
182  double stm = sin(thetaminus);
183 
184  Bkx[i][0] = ((Bkplus * ctp + Bkminus * ctm) * e1.x +
185  (-Bkplus * stp + Bkminus * stm) * e2.x) /
186  sqrt(2);
187  Bkx[i][1] = ((Bkplus * stp + Bkminus * stm) * e1.x +
188  (Bkplus * ctp - Bkminus * ctm) * e2.x) /
189  sqrt(2);
190  Bky[i][0] = ((Bkplus * ctp + Bkminus * ctm) * e1.y +
191  (-Bkplus * stp + Bkminus * stm) * e2.y) /
192  sqrt(2);
193  Bky[i][1] = ((Bkplus * stp + Bkminus * stm) * e1.y +
194  (Bkplus * ctp - Bkminus * ctm) * e2.y) /
195  sqrt(2);
196  Bkz[i][0] = ((Bkplus * ctp + Bkminus * ctm) * e1.z +
197  (-Bkplus * stp + Bkminus * stm) * e2.z) /
198  sqrt(2);
199  Bkz[i][1] = ((Bkplus * stp + Bkminus * stm) * e1.z +
200  (Bkplus * ctp - Bkminus * ctm) * e2.z) /
201  sqrt(2);
202 
203  Vector3f BkRe(Bkx[i][0], Bky[i][0], Bkz[i][0]);
204  Vector3f BkIm(Bkx[i][1], Bky[i][1], Bkz[i][1]);
205 
206  } else { // no helicity
207  if (ek.isParallelTo(n0, 1e-3_rad)) {
208  // ek parallel to (1,1,1)
209  e1.setXYZ(-1., 1., 0);
210  e2.setXYZ(1., 1., -2.);
211  } else {
212  // ek not parallel to (1,1,1)
213  e1 = n0.cross(ek);
214  e2 = ek.cross(e1);
215  }
216  e1 /= e1.getR();
217  e2 /= e2.getR();
218 
219  // random orientation perpendicular to k
220  theta = 2 * M_PI * random.rand();
221  b = e1 * cos(theta) + e2 * sin(theta);
222 
223  // normal distributed amplitude with
224  // mean = 0 and sigma = k^alpha/2
225  b *= random.randNorm() * std::pow(k, alpha / 2);
226 
227  // uniform random phase
228  phase = 2 * M_PI * random.rand();
229  cosPhase = cos(phase); // real part
230  sinPhase = sin(phase); // imaginary part
231 
232  Bkx[i][0] = b.x * cosPhase;
233  Bkx[i][1] = b.x * sinPhase;
234  Bky[i][0] = b.y * cosPhase;
235  Bky[i][1] = b.y * sinPhase;
236  Bkz[i][0] = b.z * cosPhase;
237  Bkz[i][1] = b.z * sinPhase;
238 
239  } // non helical case
240  } // for iz
241  } // for iy
242  } // for ix
243 
244  // in-place, complex to real, inverse Fourier transformation on each
245  // component note that the last elements of B(x) are unused now
246  float* Bx = (float*)Bkx;
247  fftwf_plan plan_x = fftwf_plan_dft_c2r_3d(n, n, n, Bkx, Bx, FFTW_ESTIMATE);
248  fftwf_execute(plan_x);
249  fftwf_destroy_plan(plan_x);
250

252  fftwf_plan plan_y = fftwf_plan_dft_c2r_3d(n, n, n, Bky, By, FFTW_ESTIMATE);
253  fftwf_execute(plan_y);
254  fftwf_destroy_plan(plan_y);
255 
256  float* Bz = (float*)Bkz;
257  fftwf_plan plan_z = fftwf_plan_dft_c2r_3d(n, n, n, Bkz, Bz, FFTW_ESTIMATE);
258  fftwf_execute(plan_z);
259  fftwf_destroy_plan(plan_z);
260 
261  // save to grid
262  for (size_t ix = 0; ix < n; ix++) {
263  for (size_t iy = 0; iy < n; iy++) {
264  for (size_t iz = 0; iz < n; iz++) {
265  i = ix * n * 2 * n2 + iy * 2 * n2 + iz;
266  Vector3f& b = grid->get(ix, iy, iz);
267  b.x = Bx[i];
268  b.y = By[i];
269  b.z = Bz[i];
270  }
271  }
272  }
