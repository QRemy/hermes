158 Vector3QMField JF12Field::getRegularField(const Vector3QLength &pos) const {
159  Vector3d b(0.);
160 
161  QLength r = sqrt(pos.x * pos.x + pos.y * pos.y); // in-plane radius
162  QLength d = pos.getR(); // distance to galactic center
163  if ((d < 1_kpc) or (d > 20_kpc))
164  return Vector3QMField(b); // 0 field for d < 1 kpc or d > 20 kpc
165 
166  QAngle phi = pos.getPhi(); // azimuth
167  QNumber sinPhi = sin(phi);
168  QNumber cosPhi = cos(phi);
169 
170  QNumber lfDisk = logisticFunction(pos.z, hDisk, wDisk);
171 
172  // disk field
173  if (r > 3_kpc) {
174  double bMag;
175  if (r < 5_kpc) {
176  // molecular ring
177  bMag = static_cast<double>(bRing * (5_kpc / r) *
178  (QNumber(1) - lfDisk));
179  b.x += -bMag * static_cast<double>(sinPhi);
180  b.y += bMag * static_cast<double>(cosPhi);
181 
182  } else {
183  // spiral region
184  QLength r_negx =
185  r * exp((-(phi - QAngle(pi)) / tan90MinusPitch) / 1_rad);
186  if (r_negx > rArms[7]) {
187  r_negx =
188  r * exp((-(phi + QAngle(pi)) / tan90MinusPitch) / 1_rad);
189  }
190  if (r_negx > rArms[7]) {
191  r_negx = r * exp((-(phi + QAngle(3 * pi)) / tan90MinusPitch) /
192  1_rad);
193  }
194 
195  for (int i = 7; i >= 0; i--) {
196  if (r_negx < rArms[i]) bMag = static_cast<double>(bDisk[i]);
197  }
198 
199  bMag *= static_cast<double>((5_kpc / r) * (QNumber(1) - lfDisk));
200  b.x += bMag *
201  static_cast<double>(sinPitch * cosPhi - cosPitch * sinPhi);
202  b.y += bMag *
203  static_cast<double>(sinPitch * sinPhi + cosPitch * cosPhi);
204  }
205  }
206 
207  // toroidal halo field
208  double bMagH = static_cast<double>(exp(-fabs(pos.z) / z0) * lfDisk);
209  if (pos.z >= 0_m) {
210  bMagH *= static_cast<double>(
211  bNorth * (QNumber(1) - logisticFunction(r, rNorth, wHalo)));
212  } else {
213  bMagH *= static_cast<double>(
214  bSouth * (QNumber(1) - logisticFunction(r, rSouth, wHalo)));
215  }
216  b.x += -bMagH * static_cast<double>(sinPhi);
217  b.y += bMagH * static_cast<double>(cosPhi);
218 
219  // poloidal halo field
220  double bMagX;
221  QNumber sinThetaX, cosThetaX;
222  QLength rp;
223  QLength rc = rXc + fabs(pos.z) / tanThetaX0;
224  if (r < rc) {
225  // varying elevation region
226  rp = r * rXc / rc;
227  bMagX = static_cast<double>(bX * exp(-1 * rp / rX) * pow<2>(rXc / rc));
228  QAngle thetaX = atan2(fabs(pos.z), (r - rp));
229  if (pos.z == 0_m) thetaX = pi / 2.;
230  sinThetaX = sin(thetaX);
231  cosThetaX = cos(thetaX);
232  } else {
233  // constant elevation region
234  rp = r - fabs(pos.z) / tanThetaX0;
235  bMagX = static_cast<double>(bX * exp(-rp / rX) * (rp / r));
236  sinThetaX = sinThetaX0;
237  cosThetaX = cosThetaX0;
238  }
239  double zsign = pos.z < 0_m ? -1 : 1;
240  b.x += zsign * bMagX * static_cast<double>(cosThetaX * cosPhi);
241  b.y += zsign * bMagX * static_cast<double>(cosThetaX * sinPhi);
242  b.z += bMagX * static_cast<double>(sinThetaX);
243 
244  return Vector3QMField(b);
245 }

252 QMField JF12Field::getTurbulentStrength(const Vector3QLength &pos) const {
253  if (pos.getR() > 20_kpc) return QMField(0);
254 
255  QLength r = sqrt(pos.x * pos.x + pos.y * pos.y); // in-plane radius
256  QAngle phi = pos.getPhi(); // azimuth
257 
258  // disk
259  QMField bDisk = 0;
260  if (r < 5_kpc) {
261  bDisk = bDiskTurb5;
262  } else {
263  // spiral region
264  QLength r_negx = r * exp(-(phi - QAngle(pi)) / tan90MinusPitch / 1_rad);
265  if (r_negx > rArms[7]) {
266  r_negx = r * exp(-(phi + QAngle(pi)) / tan90MinusPitch / 1_rad);
267  }
268  if (r_negx > rArms[7]) {
269  r_negx = r * exp(-(phi + QAngle(3 * pi)) / tan90MinusPitch / 1_rad);
270  }
271 
272  for (int i = 7; i >= 0; i--) {
273  if (r_negx < rArms[i]) bDisk = bDiskTurb[i];
274  }
275 
276  bDisk = bDisk * (5_kpc) / r;
277  }
278  bDisk = bDisk * exp(-0.5 * pow<2>(pos.z / zDiskTurb));
279 
280  // halo
281  QMField bHalo =
282  bHaloTurb * exp(-r / rHaloTurb) * exp(-0.5 * pow<2>(pos.z / zHaloTurb));
283 
284  // modulate turbulent field
285  return sqrt(pow<2>(bDisk) + pow<2>(bHalo));
286 }
